obj = bpy.data.objects.get(planet_name)
if obj:
    # Set orbital keyframes
    for frame in range(0, 250):
        angle = (frame / 250) * 2 * math.pi
        obj.location.x = distance_from_sun * math.cos(angle)
        obj.location.y = distance_from_sun * math.sin(angle)
        obj.keyframe_insert(data_path="location", frame=frame)
    print(f"{planet_name} animation set successfully.")
else:
    print(f"Object '{planet_name}' not found in scene.")
