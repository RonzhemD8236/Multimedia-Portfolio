import bpy
import os
import math

TEXTURE_PATH = "add texture path here (e.g. C:/Users/user name/Downloads/sun_texture.jpg)"
OUTPUT_PATH  = "add output path here (e.g. C:/Users/user name/Downloads/sun_animation.mp4)"

AXIAL_TILT_DEGREES = 7.25

TOTAL_FRAMES = 480
FPS          = 24

bpy.ops.object.select_all(action='SELECT')
bpy.ops.object.delete(use_global=False)

for block in bpy.data.meshes:    bpy.data.meshes.remove(block)
for block in bpy.data.materials: bpy.data.materials.remove(block)
for block in bpy.data.images:    bpy.data.images.remove(block)
for block in bpy.data.lights:    bpy.data.lights.remove(block)

bpy.ops.mesh.primitive_uv_sphere_add(
    segments=128,
    ring_count=64,
    radius=2.0,
    location=(0, 0, 0)
)
sun_obj = bpy.context.active_object
sun_obj.name = "Sun"

bpy.ops.object.shade_smooth()

mat = bpy.data.materials.new(name="Sun_Material")
mat.use_nodes = True
nodes = mat.node_tree.nodes
links = mat.node_tree.links
nodes.clear()

output = nodes.new("ShaderNodeOutputMaterial")
output.location = (1000, 0)

emission = nodes.new("ShaderNodeEmission")
emission.location = (650, 0)
emission.inputs["Strength"].default_value = 6.0
links.new(emission.outputs["Emission"], output.inputs["Surface"])

tex_coord = nodes.new("ShaderNodeTexCoord")
tex_coord.location = (-700, 0)

mapping = nodes.new("ShaderNodeMapping")
mapping.location = (-500, 0)
mapping.inputs["Scale"].default_value = (1.0, 1.0, 1.0)
links.new(tex_coord.outputs["UV"], mapping.inputs["Vector"])

img_node = nodes.new("ShaderNodeTexImage")
img_node.location = (-250, 150)
img_node.label = "Sun Color Map"

abs_path = bpy.path.abspath(TEXTURE_PATH)
if os.path.exists(abs_path):
    img = bpy.data.images.load(abs_path)
    img_node.image = img
    print(f"[Sun] Texture loaded: {abs_path}")
else:
    print(f"[Sun] WARNING – texture not found at: {abs_path}")

links.new(mapping.outputs["Vector"], img_node.inputs["Vector"])

mix_color = nodes.new("ShaderNodeMix")
mix_color.location = (200, 100)
mix_color.data_type = 'RGBA'
mix_color.blend_type = "MULTIPLY"
mix_color.inputs["Factor"].default_value = 0.35
mix_color.inputs["B"].default_value = (1.0, 0.55, 0.1, 1.0)
links.new(img_node.outputs["Color"], mix_color.inputs["A"])
links.new(mix_color.outputs["Result"], emission.inputs["Color"])

noise1 = nodes.new("ShaderNodeTexNoise")
noise1.location = (-250, -200)
noise1.label = "Solar Granules"
noise1.inputs["Scale"].default_value      = 12.0
noise1.inputs["Detail"].default_value     = 10.0
noise1.inputs["Roughness"].default_value  = 0.65
noise1.inputs["Distortion"].default_value = 0.4
links.new(mapping.outputs["Vector"], noise1.inputs["Vector"])

noise2 = nodes.new("ShaderNodeTexNoise")
noise2.location = (-250, -400)
noise2.label = "Fine Turbulence"
noise2.inputs["Scale"].default_value      = 30.0
noise2.inputs["Detail"].default_value     = 15.0
noise2.inputs["Roughness"].default_value  = 0.8
noise2.inputs["Distortion"].default_value = 0.2
links.new(mapping.outputs["Vector"], noise2.inputs["Vector"])

mix_noise = nodes.new("ShaderNodeMix")
mix_noise.location = (50, -300)
mix_noise.data_type = 'RGBA'
mix_noise.blend_type = "MIX"
mix_noise.inputs["Factor"].default_value = 0.45
links.new(noise1.outputs["Fac"], mix_noise.inputs["A"])
links.new(noise2.outputs["Fac"], mix_noise.inputs["B"])

ramp = nodes.new("ShaderNodeValToRGB")
ramp.location = (250, -300)
ramp.label = "Emission Variation Ramp"
ramp.color_ramp.elements[0].position = 0.3
ramp.color_ramp.elements[0].color    = (0.85, 0.4, 0.05, 1.0)
ramp.color_ramp.elements[1].position = 0.75
ramp.color_ramp.elements[1].color    = (1.0, 0.85, 0.4, 1.0)
links.new(mix_noise.outputs["Result"], ramp.inputs["Fac"])

mix_final = nodes.new("ShaderNodeMix")
mix_final.location = (450, 0)
mix_final.data_type = 'RGBA'
mix_final.blend_type = "OVERLAY"
mix_final.inputs["Factor"].default_value = 0.25
links.new(mix_color.outputs["Result"], mix_final.inputs["A"])
links.new(ramp.outputs["Color"],       mix_final.inputs["B"])
links.new(mix_final.outputs["Result"], emission.inputs["Color"])

sun_obj.data.materials.append(mat)

bpy.ops.object.light_add(type='POINT', location=(0, 0, 0))
core_light = bpy.context.active_object
core_light.name = "Sun_Core_Light"
core_light.data.energy           = 5000.0
core_light.data.color            = (1.0, 0.9, 0.6)
core_light.data.shadow_soft_size = 2.0

bpy.ops.object.light_add(type='AREA', location=(0, -6, 3))
rim = bpy.context.active_object
rim.name = "Corona_Rim"
rim.data.energy    = 80.0
rim.data.size      = 10.0
rim.data.color     = (1.0, 0.75, 0.3)
rim.rotation_euler = (1.57, 0, 0)

HDR_PATH = "add HDR path here (e.g. C:/Users/user name/Downloads/space_hdr.hdr)"

world = bpy.data.worlds["World"]
world.use_nodes = True
world_nodes = world.node_tree.nodes
world_links = world.node_tree.links
world_nodes.clear()

output_world = world_nodes.new("ShaderNodeOutputWorld")
output_world.location = (900, 0)

bg_ambient = world_nodes.new("ShaderNodeBackground")
bg_ambient.location = (300, 100)
bg_ambient.inputs["Color"].default_value    = (0.02, 0.02, 0.05, 1.0)
bg_ambient.inputs["Strength"].default_value = 0.3

img_tex = world_nodes.new("ShaderNodeTexImage")
img_tex.location  = (-200, -150)
img_tex.extension = 'EXTEND'

abs_hdr_path = bpy.path.abspath(HDR_PATH)
if os.path.exists(abs_hdr_path):
    hdr_img = bpy.data.images.load(abs_hdr_path)
    hdr_img.colorspace_settings.name = 'sRGB'
    img_tex.image = hdr_img
    print(f"[World] Background image loaded: {abs_hdr_path}")
else:
    print(f"[World] WARNING – image not found at: {abs_hdr_path}")

tex_coord_world = world_nodes.new("ShaderNodeTexCoord")
tex_coord_world.location = (-600, -150)

mapping_world = world_nodes.new("ShaderNodeMapping")
mapping_world.location = (-400, -150)
mapping_world.inputs["Scale"].default_value = (1.0, 1.0, 1.0)

world_links.new(tex_coord_world.outputs["Window"], mapping_world.inputs["Vector"])
world_links.new(mapping_world.outputs["Vector"],   img_tex.inputs["Vector"])

bg_image = world_nodes.new("ShaderNodeBackground")
bg_image.location = (300, -150)
bg_image.inputs["Strength"].default_value = 1.0
world_links.new(img_tex.outputs["Color"], bg_image.inputs["Color"])

light_path = world_nodes.new("ShaderNodeLightPath")
light_path.location = (300, 300)

mix_shader = world_nodes.new("ShaderNodeMixShader")
mix_shader.location = (650, 0)

world_links.new(light_path.outputs["Is Camera Ray"], mix_shader.inputs["Fac"])
world_links.new(bg_ambient.outputs["Background"],    mix_shader.inputs[1])
world_links.new(bg_image.outputs["Background"],      mix_shader.inputs[2])
world_links.new(mix_shader.outputs["Shader"],        output_world.inputs["Surface"])

bpy.ops.object.camera_add(location=(0, -12, 0))
cam = bpy.context.active_object
cam.name = "Camera"
cam.rotation_euler    = (1.5708, 0.0, 0.0)
cam.data.lens         = 50

bpy.context.scene.camera = cam

scene = bpy.context.scene
scene.render.engine        = 'CYCLES'
scene.cycles.device        = 'GPU'
scene.cycles.samples       = 128
scene.cycles.use_denoising = True
scene.view_layers[0].cycles.use_denoising          = True
scene.view_layers[0].cycles.denoising_store_passes = False
scene.cycles.denoiser = 'OPENIMAGEDENOISE'

scene.render.resolution_x          = 1920
scene.render.resolution_y          = 1080
scene.render.resolution_percentage = 100

scene.view_settings.view_transform = 'Filmic'
scene.view_settings.look           = 'High Contrast'
scene.view_settings.exposure       = 0.3
scene.view_settings.gamma          = 1.0

scene.render.image_settings.file_format = 'FFMPEG'
ffmpeg = scene.render.ffmpeg
ffmpeg.format               = 'MPEG4'
ffmpeg.codec                = 'H264'
ffmpeg.constant_rate_factor = 'HIGH'
ffmpeg.ffmpeg_preset        = 'GOOD'
ffmpeg.audio_codec          = 'NONE'

scene.render.filepath = OUTPUT_PATH

scene.frame_start = 1
scene.frame_end   = TOTAL_FRAMES
scene.render.fps  = FPS

bpy.context.view_layer.objects.active = sun_obj
sun_obj.select_set(True)

axial_tilt_radians = math.radians(AXIAL_TILT_DEGREES)

print("\n" + "=" * 60)
print("SUN ROTATION ANIMATION — SEAMLESS LOOP")
print("=" * 60)
print(f"Total frames         : {TOTAL_FRAMES}")
print(f"Frame rate           : {FPS} fps")
print(f"Animation duration   : {TOTAL_FRAMES / FPS:.1f} seconds")
print(f"Degrees traveled     : 360° (full seamless loop)")
print(f"Speed vs Earth       : 2.0× slower (480 vs 240 frames)")
print(f"Axial tilt           : {AXIAL_TILT_DEGREES}°")
print("=" * 60 + "\n")

for frame_num in range(scene.frame_start, scene.frame_end + 1):
    scene.frame_set(frame_num)

    progress = (frame_num - 1) / TOTAL_FRAMES
    total_rotation_radians = math.radians(progress * 360.0)

    sun_obj.rotation_euler = (axial_tilt_radians, 0, total_rotation_radians)

    sun_obj.keyframe_insert(data_path="rotation_euler", index=0)
    sun_obj.keyframe_insert(data_path="rotation_euler", index=2)

for fc in sun_obj.animation_data.action.fcurves:
    for kp in fc.keyframe_points:
        kp.interpolation = 'LINEAR'

scene.frame_set(1)

print("✓ Keyframes inserted with LINEAR interpolation (no easing pauses)")
print(f"✓ Output: {OUTPUT_PATH}")
print("\n" + "=" * 60)
print("TO RENDER THE VIDEO:")
print("=" * 60)
print("  Render menu → Render Animation  (or Ctrl+F12)")
print(f"  Output will be saved to: {OUTPUT_PATH}")
print("=" * 60 + "\n")
